          // DAETRIUS / DAYDRM — Apps Script Backend
// Paste into: Extensions → Apps Script → Code.gs
// Deploy as: Web App | Execute as: Me | Access: Anyone

function doGet(e) {
  const action = (e.parameter && e.parameter.action) || '';
  if (action === 'readAll') return readAll();
  return respond({ status: 'ok', message: 'DAETRIUS OS — backend running' });
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    if (body.action === 'writeAll') return writeAll(body.data);
    return respond({ status: 'error', message: 'Unknown action' });
  } catch(err) {
    return respond({ status: 'error', message: err.toString() });
  }
}

function readAll() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('os_data');
  if (!sheet) return respond({ status: 'error', message: 'Run setupSheets() first' });
  const val = sheet.getRange('A2').getValue();
  return respond({ status: 'ok', data: val ? JSON.parse(val) : {} });
}

function writeAll(data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('os_data');
  if (!sheet) sheet = ss.insertSheet('os_data');
  sheet.getRange('A1').setValue('data_json');
  sheet.getRange('B1').setValue('last_updated');
  sheet.getRange('A2').setValue(JSON.stringify(data));
  sheet.getRange('B2').setValue(new Date().toISOString());
  return respond({ status: 'ok', saved: true });
}

function respond(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

// Run this ONCE after pasting — creates all sheet tabs
function setupSheets() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  ['os_data','content_calendar','pipeline','tasks',
   'podcast_crm','budget','kpis','decisions'].forEach(name => {
    if (!ss.getSheetByName(name)) ss.insertSheet(name);
  });
  const s = ss.getSheetByName('os_data');
  s.getRange('A1').setValue('data_json');
  s.getRange('B1').setValue('last_updated');
  Logger.log('✅ Setup complete — now Deploy as Web App');
}
